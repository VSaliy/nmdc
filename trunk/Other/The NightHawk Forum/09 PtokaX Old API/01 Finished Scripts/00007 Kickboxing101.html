function FightCmd(curUser,cmd,hit)
-- Get cmd''s, and set C
if cmd == "kick" then C = "K"
elseif cmd == "box" then C = "B"
end
-- Get hit and set H
if hit == "low" then H = "L"
elseif hit == "med" then H = "M"
elseif hit == "high" then H = "H"
else
curUser:SendData(tConfig.Bot,"Syntax: !" ..cmd.. " <low/med/high>") return 1
end
-- Set how user choose to fight
for i,_ in pairs(KickBox["Fighters"]) do
for nick,_ in pairs(KickBox["Fighters"][i]) do
if nick == curUser.sName then
KickBox["Fighters"][i][curUser.sName][2] = C..H
SaveToFile(tConfig.File,KickBox,"KickBox")
FightReady(i,curUser) -- Send fight nr
end
end
end
end

function WinLose(fight)
a,b = nil,nil
for nick,_ in pairs(KickBox["Fighters"][fight]) do
if a then b = nick else a = nick end -- get nick
end
local win = { a,b } -- choose winner
winner = win[math.random(table.maxn(win))]
if a == winner then looser = b else looser = a end
Fight(winner,looser,fight)
end
function Fight(win,lose,fight)

high = {"head","nose","forehead"}
low = {"crotch","knee","leg"}
local _,_,wD,wT = string.find(KickBox["Fighters"][fight][win][2],"(%a)(%a)")
if wD == "K" then wDo = "kicks" else wDo = "boxes" end
if wT == "L" then
wTar = low[math.random(table.maxn(low))]
elseif wT == "M" then
wTar = mid[math.random(table.maxn(mid))]
elseif wT == "H" then
wTar = high[math.random(table.maxn(high))]
end
local _,_,lD,lT = string.find(KickBox["Fighters"][fight][lose][2],"(%a)(%a)")
if lD == "K" then lDo = "kick" else lDo = "boxes" end
if lT == "L" then
lTar = low[math.random(table.maxn(low))]
wAvoid = "jumps"
elseif lT == "M" then
lTar = mid[math.random(table.maxn(mid))]
wAvoid = "step''s to the left"
elseif lT == "H" then
lTar = high[math.random(table.maxn(high))]
wAvoid = "ducks"
end
Msg = { win.. " " ..wDo.. " " ..lose.. " in the " ..wTar, lose.. " tries " ..lDo.. " " ..win.. " in the " ..lTar.. ", but " ..win.. " "..wAvoid.. " and then " ..win.. " " ..wDo.. " " ..lose.. " in the " ..wTar }

end

function Serialize(tTable, sTableName, sTab)
assert(tTable, "tTable equals nil");
assert(sTableName, "sTableName equals nil");
assert(type(tTable) == "table", "tTable must be a table!");
assert(type(sTableName) == "string", "sTableName must be a string!");
sTab = sTab or "";
sTmp = ""
sTmp = sTmp..sTab..sTableName.." = {
"
for key, value in pairs(tTable) do
local sKey = (type(key) == "string") and string.format("[%q]",key) or string.format("[%d]",key);
if(type(value) == "table") then
sTmp = sTmp..Serialize(value, sKey, sTab.."");
else
local sValue = (type(value) == "string") and string.format("%q",value) or tostring(value);
sTmp = sTmp..sTab..""..sKey.." = "..sValue
end
sTmp = sTmp..",
"
end
sTmp = sTmp..sTab.."}"
return sTmp
end
function SaveToFile(file , table , tablename)
local handle = io.open(file,"w+")
handle:write(Serialize(table, tablename))
handle:flush()
handle:close()
end
function LoadFromFile(filename)
local f = io.open(filename)
if f then
local r = f:read("*a")
f:flush()
f:close()
local func,err = loadstring(r)
if func then x,err = pcall(func) end
end
end
